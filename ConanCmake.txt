cmake_minimum_required(VERSION 3.10)
project(cmtj_server)

set(CMAKE_CXX_STANDARD 17)

include(${CMAKE_BINARY_DIR}/conan_paths.cmake)

find_library(
        fftw3
        NAMES fftw lfftw3 libfftw3
        HINTS /usr/local/lib /usr/local/include
)

find_package(Boost)
find_package(nlohmann_json 3.2.0 REQUIRED)
find_package (Threads)
find_package(fmt)
find_package (leveldb REQUIRED)
find_package(httplib REQUIRED)
find_package(spdlog REQUIRED)
find_package(OpenSSL REQUIRED)
if( OPENSSL_FOUND )
    include_directories(${OPENSSL_INCLUDE_DIR})
    message(STATUS "Using OpenSSL ${OPENSSL_VERSION}")
endif()

# HEADERS for the junction 
set(LOC_DIR "/app/")
include_directories(${LOC_DIR})

add_executable(cmtj_server
              main.cpp
              ${LOC_DIR}/junction.hpp
              ${LOC_DIR}/drivers.hpp
              ${LOC_DIR}/cvector.hpp)
target_link_libraries(cmtj_server PRIVATE spdlog::spdlog nlohmann_json::nlohmann_json httplib::httplib leveldb::leveldb fftw3 fftwpp)
